#include<bits/stdc++.h>
using namespace std;


void solve()
{
    int n , m ;
    cin >> n >> m;
    int oddCount = 1;
    for(int i = 0 ; i < n ; i++){

        if(i % 2 == 0){
           
            for(int j = 0 ; j < m ;j++){
                cout << '#';
            }
            cout << endl;
        }
        else{
            if(oddCount % 2 == 1){
                for(int j = 0 ; j < m - 1 ; j++){
                    cout << '.';
                }

                cout << "#";
                cout << endl;
                oddCount++;
            }
            else{
                cout << '#';
                for(int j = 0 ; j < m - 1 ; j++){
                    cout << '.';
                }
                cout << endl;
                oddCount++;
            }
        }

        
    }
}
int main(){
    solve();
    return 0;
}
