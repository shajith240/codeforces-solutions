#include<bits/stdc++.h>
using namespace std;

int main(){
    int n ;
    cin >> n;
    vector<int> a(n);
    int totalsum = 0;
    for(int i = 0 ; i < n ; i++){
        cin >> a[i];
        totalsum += a[i];
    }
    
    sort(a.rbegin(),a.rend());
    int mySum = 0;
    int count = 0;
    int i = 0;
    while(mySum <= (totalsum - mySum)){
        mySum += a[i];
        count++;
        i++;
    }
    
    cout << count << endl;
    return 0;
}
