#include <bits/stdc++.h>
using namespace std;

bool is_dangerous(string s) {
    int count = 1;   // start from 1

    for (int i = 1; i < s.length(); i++) {
        if (s[i] == s[i - 1]) {
            count++;
            if (count == 7)
                return true;
        } else {
            count = 1;
        }
    }

    return false;
}

void solve() {
    string s;
    cin >> s;

    if (is_dangerous(s)) {
        cout << "YES" << endl;
    } else {
        cout << "NO" << endl;
    }
}

int main() {
    solve();
    return 0;
}
