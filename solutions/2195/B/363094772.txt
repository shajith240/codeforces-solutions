#include<bits/stdc++.h>
using namespace std;

int g_odd(int n){
    
    while(n % 2 == 0){
        n  =  n / 2;
    }
    return n;
}

void solve()
{
    int n;
    cin >> n;
    int a[n];
    for(int i = 0 ; i < n ;i++){
        cin >> a[i];
    }

    for(int i = 0 ; i < n ; i++){
        if(g_odd(i + 1) != g_odd(a[i])){
            cout << "NO" << endl;
            return;
        }
    }
    cout << "YES"<< endl;
  
}

int main()
{   
    int t;
    cin >> t;
    while(t--){
        solve();
    }
    return 0;
}
