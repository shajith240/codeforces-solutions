#include<bits/stdc++.h>
using namespace std;


void solve(){
    int n;
    cin >> n;
    vector<int> a(n);
    bool flag = false;

    for(int i = 0 ; i < n ; i++){
        cin >> a[i];
        if(a[i] == 67) {
            flag= true;
        }
    }

    if(flag) {
        cout << "YES" << endl;
    } else {
        cout << "NO" << endl;
    }
}

int main(){
    
    int t;
    cin >> t;
    while(t--){
       solve();
    }
    
    return 0;
}
