#include<bits/stdc++.h>
using namespace std;

int main(){
    int k ;
    cin >> k;
    vector<int> a(12);
    
    for(int i = 0 ; i < 12 ; i++){
        cin >> a[i];
    }
    
    sort(a.rbegin(),a.rend());

    int currentgrowth = 0;
    int count = 0;
    int i = 0;
    while(currentgrowth < k && i < 12){
        currentgrowth += a[i];
        count++;
        i++;
    }
    
    
    if(currentgrowth < k){
        cout << -1 << endl;
    }
    else{
        cout << count << endl;
    }
    return 0;
}
